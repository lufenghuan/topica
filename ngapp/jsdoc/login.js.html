/**
* @doc overview
* @id login_controller
* @name topicaApp.controllers:LoginController
* @description The login controller.
*
* The controller of the login view.
* View calls submit funtion to login. $scope.credential is bind to the email input and password input.
* When login is successful, call loginConfirmed method authService module.
* A login confirmed event will be broadcasted.
*
* Route: /login
*
*/


(function() {
  var app;

  app = angular.module("topicaApp");

  app.controller("LoginController", [
    'Configs', '$http', '$scope', 'authService', 'UserSession', function(Configs, $http, $scope, authService, UserSession) {
      $scope.credential = {
        email: "",
        password: "",
        rememberMe: false
      };
      return $scope.submit = function() {
        return UserSession.login($scope.credential.email, $scope.credential.password, $scope.credential.rememberMe).then(function(response) {
          var user;
          user = response.data.user;
          return authService.loginConfirmed(user);
        }, function(error) {
          return $scope.message = "Invalid username or password.";
        });
      };
    }
  ]);

}).call(this);
