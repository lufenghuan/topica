def create
  if @user = login(params[:email], params[:password], params[:remember])
    render json: {:ok => true, :user => {:id => @user.id, :username => @user.username, :email => @user.email}}
  else
    render json: {:ok => false}, status: :unprocessable_entity
  end
end
